```html
<body>
  <main></main>

  <div id="title">
    <!-- Title-->
  </div>
  <div id="subTitle">
    <!-- Subtitle -->
  </div>
  <div id="copy">
    <!-- Footer -->
  </div>

  <script src="https://cdn.jsdelivr.net/gh/katzEco/legacy-code@main/js-gen-updated/main.js"></script>

```
